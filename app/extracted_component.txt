            Every resource a student, licensed barber, shop owner, or school needs —
            ranked, reviewed, and ready. Updated for 2025.
          </p>

          {/* Stats bar */}
          <div style={{ display: "flex", gap: "40px", flexWrap: "wrap" }}>
            {[
              { num: `${TOOLS.length}+`, label: "Curated Resources" },
              { num: "6", label: "Categories" },
              { num: "4", label: "Audience Types" },

                <div style={{ fontSize: "12px", color: "#555", fontFamily: "'DM Mono', monospace", textTransform: "uppercase", letterSpacing: "0.08em" }}>{s.label}</div>
              </div>
            ))}
          </div>
        </div>
      </div>

      {/* Controls */}
      <div style={{
        maxWidth: "1200px",
        margin: "0 auto",
        padding: "0 40px 32px",
      }}>
        {/* Category pills */}
        <div style={{
          display: "flex",
          gap: "8px",
          overflowX: "auto",
          paddingBottom: "4px",
          marginBottom: "20px",
        }}>
          {CATEGORIES.map((c) => (
            <button
              key={c.id}
              className={`cat-btn ${activeCategory === c.id ? "active" : ""}`}
              onClick={() => setActiveCategory(c.id)}
            >
              {c.icon} {c.label}
            </button>
          ))}
        </div>

        {/* Audience + Search row */}

          <div style={{ display: "flex", gap: "6px", flexWrap: "wrap" }}>
            {AUDIENCES.map((a) => (
              <button
                key={a}
                className={`aud-btn ${activeAudience === a ? "active" : ""}`}
                onClick={() => setActiveAudience(a)}
              >

              </button>
            ))}
          </div>
          <input
            className="search-input"
            placeholder="Search tools..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
          />
        </div>
      </div>

      {/* Tools Grid */}
      <div style={{ maxWidth: "1200px", margin: "0 auto", padding: "0 40px 80px" }}>
        {filtered.length === 0 ? (

            <div style={{ fontSize: "48px", marginBottom: "16px" }}>✂️</div>
            <div style={{ fontFamily: "'Syne', sans-serif", fontSize: "20px" }}>No tools found for that filter.</div>
          </div>
        ) : (
          <div className="grid">
            {filtered.map((tool) => (
              <ToolCard key={tool.id} tool={tool} />
            ))}
          </div>
        )}
      </div>

      {/* Footer note */}
      <div style={{
        borderTop: "1px solid #161616",
        padding: "32px 40px",
        maxWidth: "1200px",
        margin: "0 auto",
        display: "flex",
        justifyContent: "space-between",
        alignItems: "center",
        flexWrap: "wrap",
        gap: "12px",
      }}>

          <br />All tools are independently reviewed and ranked on merit.
        </div>
        <div style={{ fontFamily: "'Syne', sans-serif", fontSize: "13px", color: "#C8A96E", fontWeight: 700 }}>
          FadeJunkie © 2025
        </div>
      </div>
    </div>
  );
}

function ToolCard({ tool }) {
  const [imgError, setImgError] = useState(false);
  const rankColor = RANK_COLORS[tool.rank] || "#444";

  return (
    <div
      className={`tool-card ${tool.featured ? "featured" : ""}`}
      onClick={() => window.open(tool.url, "_blank")}
    >
      {/* Top row: logo + name + rank */}
      <div style={{ display: "flex", alignItems: "flex-start", gap: "14px" }}>
        {!imgError ? (
          <img
            src={tool.logo}
            alt={tool.name}
            className="logo-img"
            onError={() => setImgError(true)}
          />
        ) : (
          <div className="logo-fallback">✂️</div>
        )}
        <div style={{ flex: 1, minWidth: 0 }}>
          <div style={{ display: "flex", alignItems: "center", gap: "8px", marginBottom: "4px" }}>
            <span style={{
              fontFamily: "'Syne', sans-serif",
              fontWeight: 700,
              fontSize: "16px",
              lineHeight: 1.2,
              color: tool.featured ? "#C8A96E" : "#F5F0E8",
            }}>{tool.name}</span>
            {tool.featured && (
              <span style={{
                fontSize: "10px",
                background: "#C8A96E",
                color: "#0A0A0A",
                padding: "2px 7px",
                borderRadius: "4px",
                fontWeight: 800,
                fontFamily: "'DM Mono', monospace",
                textTransform: "uppercase",
                letterSpacing: "0.05em",
              }}>Our Platform</span>
            )}
          </div>
          <div style={{ display: "flex", gap: "6px", flexWrap: "wrap" }}>
            <span className="badge-pill">{tool.badge}</span>
            {tool.affiliate && <span className="affiliate-chip">affiliate</span>}
          </div>
        </div>
        {/* Rank indicator */}
        {!tool.featured && (
          <div style={{
            fontFamily: "'DM Mono', monospace",
            fontSize: "11px",
            color: rankColor,
            fontWeight: 500,
            flexShrink: 0,
          }}>
            #{tool.rank}
          </div>
        )}
      </div>

      {/* Tagline */}
      <div style={{
        fontFamily: "'Syne', sans-serif",
        fontSize: "13px",
        fontWeight: 600,
        color: "#AAA",
        lineHeight: 1.4,
      }}>
        {tool.tagline}
      </div>

      {/* Description */}
      <div style={{
        fontSize: "13px",
        color: "#666",
        lineHeight: 1.65,
        flexGrow: 1,
      }}>
        {tool.description}
      </div>

      {/* Price + audiences row */}
      <div style={{
        display: "flex",
        justifyContent: "space-between",
        alignItems: "center",
        flexWrap: "wrap",
        gap: "8px",
      }}>
        <div style={{
          fontFamily: "'DM Mono', monospace",
          fontSize: "11px",
          color: "#C8A96E",
          fontWeight: 500,
        }}>
          {tool.price}
        </div>
        <div style={{ display: "flex", gap: "4px", flexWrap: "wrap" }}>
          {tool.audiences.map((a) => (
            <span key={a} className="audience-tag">{a}</span>
          ))}
        </div>
      </div>

      {/* CTA */}
      <a
        href={tool.url}
        target="_blank"
        rel="noopener noreferrer"
        className={`visit-btn ${tool.affiliate ? "affiliate" : ""}`}
        onClick={(e) => e.stopPropagation()}
      >
        {tool.featured ? "Join FadeJunkie →" : `Visit ${tool.name.split(" ")[0]} →`}
      </a>
    </div>
  );
}
